
// BASE NODE
Block = List Statement
Statement = VariableDeclaration + EOI
Statement = FunctionDeclaration + EOI
Statement = ConditionalBranch + EOI
Statement = Assignment + EOI

// VARIABLE DECLARATION
VariableDeclaration = VAR + SYMBOL
VariableDeclaration = VAR + SYMBOL + ASSIGN + Value

// ASSIGNMENT
Assignment = SYMBOL + ASSIGN + Value

// FUNCTION DECLARATION
FunctionDeclaration = FUN + SYMBOL + LPAREN + List FunctionArgument + RPAREN + LBRACE + Block + RBRACE

// FUNCTION ARGUMENT
FunctionArgument = SYMBOL
FunctionArgument = SYMBOL + COMMA + FunctionArgument

// CONDITIONAL BRANCH
ConditionalBranch = IF + LPAREN + Condition + RPAREN + LBRACE + Block + RBRACE
ConditionalBranch = IF + LPAREN + Condition + RPAREN + LBRACE + Block + RBRACE + ELSE + LBRACE + Block + RBRACE

// CONDITION
Condition = Value + ConditionOperator + Value

// CONDITION OPERATOR
ConditionOperator = EQUAL
ConditionOperator = GREATER
ConditionOperator = LOWER
ConditionOperator = GREATEREQUAL
ConditionOperator = LOWEREQUAL

// OPERATION
Operation = Value + OperationOperator + Value

// OPERATOR
OperationOperator = ADD
OperationOperator = SUB
OperationOperator = MUL
OperationOperator = DIV
OperationOperator = MOD

// VALUE
Value = STRING
Value = INTEGER
Value = FLOAT
Value = SYMBOL
